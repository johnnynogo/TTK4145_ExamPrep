mutex = Semaphore(1)
roomEmpty = Semaphore(1)
int readers = 0

// Writer
roomEmpty.wait()
    write()
roomEmpty.signal()

// Readers
mutex.wait()
    readers++
    if readers == 1:
        roomEmpty.wait()
mutex.signal()

read()

mutex.lock()
    readers--
    if readers == 0
        roomEmpty.signal()
mutex.signal()
