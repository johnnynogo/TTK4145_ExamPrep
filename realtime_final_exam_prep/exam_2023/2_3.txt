// Exit protocols
Semaphore mutex(1);
Semaphore exitMutex(1);
Semaphore coordinatorNeeded(0);
Semaphore capacity(0);
Semaphore exitBarrier(0);
int workers = 0;
int totalIn = 0;
int finishedCount = 0;

void workerEntry(){
    wait(mutex);
        workers++;
        if(workers % 3 == 1)
            signal(coordinatorNeeded);
    signal(mutex);
    wait(capacity);
    wait(mutex); totalIn++; signal(mutex);
}

void coordinatorEntry(){
    wait(coordinatorNeeded);
    signal(capacity);
    signal(capacity);
    signal(capacity);
    wait(mutex); totalIn++; signal(mutex);
}

void exit(){
    wait(exitMutex);
        finishedCount++;
        if(finishedCount == totalIn){
            workers = 0;
            totalIn = 0;
            finishedCount = 0;
            exitBarrier.signal();
        }
    signal(exitMutex);

    exitBarrier.wait();
    exitBarrier.signal();
}
